/**
 * @param {number[]} nums
 * @return {number[][]}
 * 
 * when input [1,2,3]
 * sholud output [[],[1],[2],[1,2],[3],[1,3],[2,3],[1,2,3]]
 */
const subsets = (nums) => {
    const res = [], path = [], len = nums.length;
    getSubset(0);
    return res;
    function getSubset(index) {
        res.push([...path]);

        for (let i = index; i < len; i++) {
            path.push(nums[i]);
            getSubset(i + 1);
            path.pop();
        }
    }
}
